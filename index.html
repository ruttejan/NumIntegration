Obd√©ln√≠kov√° metoda

Obd√©ln√≠kov√° metoda

  • Anglicky: ‚ÄúMidpoint Rule‚Äù

  • Metoda rozdƒõl√≠ integrovan√Ω interval na \(n\) stejn√Ωch podinterval≈Ø (\(n + 1\) bod≈Ø)

  • Mezi ka≈ædou dvojic√≠ po sobƒõ jdouc√≠ch bod≈Ø vytvo≈ô√≠ obd√©ln√≠k s v√Ω≈°kou funkƒçn√≠ hodnoty jejich st≈ôedu

  • Aproximace integr√°lu se tedy rovn√° souƒçtu obsah≈Ø obd√©ln√≠k≈Ø

\[h = \frac{a - b}{n}\]

\[x_0 = a; x_i = a + i \cdot h; x_n = b,\space\space \text{kde} \space\space i = 1\dots n-1\]

\[\int_{a}^{b} f(x) \, dx \approx \sum_{i=1}^{n} f\left(\frac{x_{i-1} + x_i}{2}\right) h\]

V√Ωpoƒçet - Obd√©ln√≠kov√° metoda

function numerical_int(::Midpoint,
    f::Function,
    a::Real,
    b::Real,
    n::Int
)
    h = (b - a) / n
    x_values = collect(a:h:b)
    integral = 0
    for i = 2:n
        integral += f((x_values[i-1] + x_values[i]) / 2)
    end
    return h * integral
end

Lichobƒõ≈æn√≠kov√° metoda

Lichobƒõ≈æn√≠kov√° metoda

  • Anglicky: ‚ÄúTrapezoid Rule‚Äù

  • Rozdƒõl√≠me si interval stejnƒõ jako u obd√©ln√≠kov√© metody; \(n\) podinterval≈Ø (\(n + 1\) bod≈Ø)

  • Funkƒçn√≠ hodnoty dvou po sobƒõ jdouc√≠ch bod≈Ø spoj√≠me √∫seƒçkou (vyto≈ô√≠me lichobƒõ≈æn√≠k)

  • Aproximace intgr√°lu se tedy rovn√° sumƒõ obsah≈Ø lichobƒõ≈æn√≠k≈Ø

\[h = \frac{a - b}{n}\]

\[x_0 = a;\space x_i = a + i \cdot h;\space x_n = b,\space\space \text{kde} \space\space i = 1\dots n-1\]

\[\int_{a}^{b} f(x) \, dx \approx \frac{h}{2} (x_0 + x_n + \sum_{i=1}^{n-1} 2f(x_i))\]

V√Ωpoƒçet - Lichobƒõ≈æn√≠kov√° metoda

function numerical_int(::Trapezoid,
    f::Function,
    a::Real,
    b::Real,
    n::Int
)
    h = (b - a) / n
    x_values = collect((a+h):h:(b-h))
    integral = (f(a) + f(b))
    for i = 1:length(x_values)-1
        integral += 2 * f(x_values[i])
    end
    return (h / 2) * integral
end

Simpsonova metoda

Simpsonova metoda

  • Opƒõt rozdƒõl√≠me interval na \(n\) podinterval≈Ø (\(n + 1\) bod≈Ø)

  • Funkƒçn√≠mi hodnotami krajn√≠ch bod≈Ø a st≈ôedu intrerval≈Ø prolo≈æ√≠me paraboly

  • Aproximace integr√°lu se zjednodu≈°≈°√≠ na v√Ωpoƒçet integr√°lu z kvadratick√© funkce \[h = \frac{a - b}{n}\] \[x_0 = a;\space x_i = a + i \cdot h;\space x_n = b,\space\space \text{kde} \space\space i = 1\dots n-1\] \[\int_{x_i}^{x_{i+1}} f(x) \, dx \approx \int_{x_i}^{x_{i+1}} ax^2 + bx + c \, dx = \frac{h}{6}(f(x_i) + 4f(x_i + \frac{h}{2}) + f(x_{i+1})) \]

  • Ze vzoreƒçku je vidƒõt, ≈æe krajn√≠ body maj√≠ v√°hu 1 a st≈ôedov√© body maj√≠ v√°hu 4.

  • Jeliko≈æ v≈°echny krajn√≠ body podinterval≈Ø (kromƒõ krajn√≠ch bod≈Ø p≈Øvodn√≠ho intervalu) pou≈æijeme dvakr√°t, p≈ôi≈ôad√≠me jim v√°hu 2 (pro v√≠ce informac√≠: Navara 2023) \[\int_{a}^{b} f(x) \, dx \approx \frac{h}{6}(x_0 + x_n + \sum_{i=1}^{n-1} 2f(x_i) + \sum_{i=0}^{n-1} 4f(x_i + \frac{h}{2}))\]

V√Ωpoƒçet - Simpsonova metody

function numerical_int(::Simpson,
    f::Function,
    a::Real,
    b::Real,
    n::Int
)
    h = (b - a) / n
    h2 = h / 2
    x_edge = collect((a+h):h:(b-h))
    x_mids = collect((a+h2):h:(b-h/2))
    integral = f(a) + f(b)
    for i in eachindex(x_edge)
        integral += 2 * f(x_edge[i])
    end

    for i in eachindex(x_mids)
        integral += 4 * f(x_mids[i])
    end
    return integral * (h / 6)
end

integrate()

  • Tato funkce iteruje, dokud rozd√≠l aktu√°ln√≠ a p≈ôedchoz√≠ aproximace nen√≠ men≈°√≠ ne≈æ nƒõjak√Ω zadan√Ω threshold, nebo dokud nen√≠ dosa≈æen maxim√°ln√≠ poƒçet iterac√≠.
function integrate(
    method::IntMethod,
    f::Function,
    a::Real,
    b::Real;
    threshold::Real=1e-8,
    numIter::Int=20
)

    n = 2  # Initial number of split points
    prev_integral = numerical_int(method, f, a, b, n)
    i = 0
    while i < numIter
        n *= 2  # Double the number of split points
        integral = numerical_int(method, f, a, b, n)
        difference = abs(integral - prev_integral)

        if difference < threshold
            error = error_int(method, f, a, b, n)
            return integral, error, n
        end

        prev_integral = integral
        i += 1
    end
    error = error_int(method, f, a, b, n)
    return prev_integral, error, n
end

integrate_compare()

  • Tato funkce iteruje, dokud rozd√≠l aproximace a referenƒçn√≠ aproximace z knihovny Quadgk nen√≠ men≈°√≠ ne≈æ nƒõjak√Ω zadan√Ω threshold, nebo dokud nen√≠ dosa≈æen maxim√°ln√≠ poƒçet iterac√≠.
function integrate_compare(
    method::IntMethod,
    f::Function,
    a::Real,
    b::Real;
    threshold::Real=1e-8,
    numIter::Int=20
)

    ref_result, _ = quadgk(f, a, b)
    n = 1  # Initial number of split points
    i = 0
    integral = numerical_int(method, f, a, b, n)
    while i < numIter
        n *= 2  # Double the number of split points
        integral = numerical_int(method, f, a, b, n)
        difference = abs(ref_result - integral)

        if difference < threshold
            error = error_int(method, f, a, b, n)
            return integral, error, n
        end
        i += 1
    end
    error = error_int(method, f, a, b, n)
    return integral, error, n

end

error_int()

(Menz and Mulberry, n.d.)

function error_int(::Midpoint, 
    f :: Function,
    a :: Real, 
    b :: Real, 
    n :: Int
)
    fd1(x) = ForwardDiff.derivative(f, x)
    fd2(x) = ForwardDiff.derivative(fd1, x)

    fx_max = findmax(x -> abs(fd2(x)), a:b)[1]
    return (((b - a)^3) * fx_max) / (24 * n^2)
end

function error_int(::Trapezoid,
    f::Function,
    a::Real,
    b::Real,
    n::Int
)
    fd1(x) = ForwardDiff.derivative(f, x)
    fd2(x) = ForwardDiff.derivative(fd1, x)

    fx_max = findmax(x -> abs(fd2(x)), a:b)[1]
    return (b - a)^3 / 12 * fx_max / n^2
end

function error_int(::Simpson,
    f::Function,
    a::Real,
    b::Real,
    n::Int
)
    fd1(x) = ForwardDiff.derivative(f, x)
    fd2(x) = ForwardDiff.derivative(fd1, x)
    fd3(x) = ForwardDiff.derivative(fd2, x)
    fd4(x) = ForwardDiff.derivative(fd3, x)

    fx_max = findmax(x -> abs(fd4(x)), a:b)[1]
    return (((b - a)^5) * fx_max) / (180 * n^4)
end
